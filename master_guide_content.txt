B2. Tamil Grammar ΓÇö Core Concepts
Tamil uses Subject-Object-Verb (SOV) order. Verb endings encode tense AND person/gender simultaneously.

# backend/grammar_engine.py
class TamilGrammarEngine:
    def __init__(self):
        self.present={'first':'α«òα«┐α«▒α»çα«⌐α»ì','male':'α«òα«┐α«▒α«╛α«⌐α»ì','female':'α«òα«┐α«▒α«╛α«│α»ì','neutral':'α«òα«┐α«▒α«ñα»ü','plural':'α«òα«┐α«▒α«╛α«░α»ìα«òα«│α»ì'}
        self.past   ={'first':'α«⌐α»çα«⌐α»ì','male':'α«⌐α«╛α«⌐α»ì','female':'α«⌐α«╛α«│α»ì','neutral':'α«⌐α«ñα»ü','plural':'α«⌐α«╛α«░α»ìα«òα«│α»ì'}
        self.future ={'first':'α«╡α»çα«⌐α»ì','male':'α«╡α«╛α«⌐α»ì','female':'α«╡α«╛α«│α»ì','neutral':'α«òα»üα««α»ì','plural':'α«╡α«╛α«░α»ìα«òα«│α»ì'}
        self.subjects={'first':'α«¿α«╛α«⌐α»ì','male':'α«àα«╡α«⌐α»ì','female':'α«àα«╡α«│α»ì','neutral':'α«àα«ñα»ü','plural':'α«àα«╡α«░α»ìα«òα«│α»ì'}
    def get_suffix(self,tense,gender):
        return {'present':self.present,'past':self.past,'future':self.future}.get(tense,self.present).get(gender,'α«òα«┐α«▒α»çα«⌐α»ì')
    def generate_statement(self,noun,verb,tense,gender):
        return f"{self.subjects.get(gender,'α«¿α«╛α«⌐α»ì')} {noun} {verb}{self.get_suffix(tense,gender)}"
    def generate_question(self,noun,verb,tense,gender):
        return self.generate_statement(noun,verb,tense,gender)[:-1]+'α«å?'
    def generate_dialogue(self,noun,verb,tense,gender):
        return [{'speaker':'α«¿α«úα»ìα«¬α«⌐α»ì','line':self.generate_question(noun,verb,tense,gender)},
                {'speaker':'α«¿α«╛α«⌐α»ì','line':self.generate_statement(noun,verb,tense,gender)}]

# backend/app.py
from flask import Flask,request,jsonify,render_template
from flask_cors import CORS
import sys,os; sys.path.insert(0,os.path.dirname(__file__))
from grammar_engine import TamilGrammarEngine
app=Flask(__name__,template_folder='../frontend/templates',static_folder='../frontend/static')
CORS(app); engine=TamilGrammarEngine()
@app.route('/')
def index(): return render_template('index.html')
@app.route('/generate',methods=['POST'])
def generate():
    d=request.get_json()
    n,v,t,g,s=d['noun'],d['verb'],d.get('tense','present'),d.get('gender','first'),d.get('sentence_type','statement')
    out=engine.generate_statement(n,v,t,g) if s=='statement' else \
        engine.generate_question(n,v,t,g) if s=='question' else engine.generate_dialogue(n,v,t,g)
    return jsonify({'type':s,'output':out})
if __name__=='__main__': app.run(debug=True,port=5000)

# backend/db_manager.py
import sqlite3,os
DB=os.path.join(os.path.dirname(__file__),'tamil_nlp.db')
def conn():
    c=sqlite3.connect(DB); c.row_factory=sqlite3.Row; return c
def get_all_vocabulary():
    db=conn(); r=db.execute('SELECT * FROM vocabulary ORDER BY category').fetchall()
    db.close(); return [dict(x) for x in r]
def search_vocabulary(q):
    db=conn(); r=db.execute('SELECT * FROM vocabulary WHERE tamil_word LIKE ? OR english_meaning LIKE ?',
        (f'%{q}%',f'%{q}%')).fetchall(); db.close(); return [dict(x) for x in r]
def get_suffix(tense,gender):
    db=conn(); r=db.execute('SELECT suffix FROM grammar_rules WHERE tense=? AND gender=?',(tense,gender)).fetchone()
    db.close(); return r['suffix'] if r else 'α«òα«┐α«▒α»çα«⌐α»ì'
def get_subject(gender):
    db=conn(); r=db.execute('SELECT subject FROM grammar_rules WHERE gender=? LIMIT 1',(gender,)).fetchone()
    db.close(); return r['subject'] if r else 'α«¿α«╛α«⌐α»ì'
def save_sentence(sentence,stype,tense):
    db=conn(); db.execute('INSERT INTO saved_sentences(sentence,sentence_type,tense) VALUES(?,?,?)',(sentence,stype,tense))
    db.commit(); db.close()
def get_saved():
    db=conn(); r=db.execute('SELECT * FROM saved_sentences ORDER BY saved_at DESC').fetchall()
    db.close(); return [dict(x) for x in r]
def delete_saved(sid):
    db=conn(); db.execute('DELETE FROM saved_sentences WHERE id=?',(sid,)); db.commit(); db.close()

async function speakSentence(){
  if(!last)return;
  const text=last.data.type==='dialogue'?last.data.output.map(d=>d.line).join(' '):last.data.output;
  const r=await(await fetch('/speak',{method:'POST',headers:{'Content-Type':'application/json'},

  if(r.audio_url){const a=document.getElementById('audio');
    a.src=r.audio_url+'?t='+Date.now();
    document.getElementById('audioPlayer').style.display='block'; a.play();}
}
function startVoice(){
  if(!('webkitSpeechRecognition' in window)){alert('Use Chrome/Edge for voice.');return;}
  const r=new webkitSpeechRecognition();
  r.lang='ta-IN'; r.interimResults=false;
  r.onresult=e=>{ document.getElementById('noun').value=e.results[0][0].transcript; };
  r.onerror=e=>alert('Voice error: '+e.error);
  r.start();
}

import anthropic as ant
from dotenv import load_dotenv
load_dotenv()
ai=ant.Anthropic(api_key=os.environ.get('ANTHROPIC_API_KEY'))

@app.route('/variations',methods=['POST'])
def variations():
    sent=request.get_json().get('sentence','')
    prompt=f'You are a Tamil language teacher. Sentence: {sent}. Give 3 alternative Tamil ways to say the same thing. Number each. Tamil first, English in brackets.'
    msg=ai.messages.create(model='claude-sonnet-4-20250514',max_tokens=400,
        messages=[{'role':'user','content':prompt}])
    return jsonify({'variations':msg.content[0].text})

@app.route('/chat',methods=['POST'])
def chat():
    d=request.get_json()
    msgs=d.get('history',[])+[{'role':'user','content':d.get('message','')}]
    sys='You are Amma, a friendly Tamil tutor. Reply in simple English with Tamil examples. Max 3 sentences.'
    r=ai.messages.create(model='claude-sonnet-4-20250514',max_tokens=300,system=sys,messages=msgs)
    return jsonify({'reply':r.content[0].text})

@app.route('/chatbot')
def chatbot(): return render_template('chatbot.html')

def check_grammar():
    text=request.get_json().get('text','')
    prompt=f'''You are an expert Tamil grammar teacher.
Analyse this Tamil text: {text}
1. Identify any grammar errors (tense, verb agreement, case markers).
2. Provide corrected version.
3. Explain each correction in simple English.
Format: ERROR | CORRECTION | EXPLANATION (one per line)'''

@app.route('/api/analytics/<int:uid>')
def analytics(uid):
    db=conn()
    total=db.execute('SELECT COUNT(*) as c FROM learning_sessions WHERE user_id=?',(uid,)).fetchone()['c']
    by_activity=db.execute('SELECT activity,COUNT(*) as c FROM learning_sessions WHERE user_id=? GROUP BY activity',(uid,)).fetchall()
    avg_score=db.execute('SELECT AVG(score) as s FROM quiz_results WHERE user_id=?',(uid,)).fetchone()['s']
    db.close()
    return jsonify({'total_sessions':total,'by_activity':[dict(x) for x in by_activity],'avg_quiz_score':round(avg_score or 0,1)})

@app.route('/submit_assignment',methods=['POST'])
def submit_assignment():
    d=request.get_json()
    asgn_id,student_id,text=d['assignment_id'],d['student_id'],d['text']
    prompt=f'Evaluate this Tamil sentence for grammar correctness (score 0ΓÇô10) and give brief feedback: {text}. Return JSON: {{score:int,feedback:str}}'
    r=ai.messages.create(model='claude-sonnet-4-20250514',max_tokens=200,messages=[{'role':'user','content':prompt}])

        (asgn_id,student_id,text,ev['score'],ev['feedback']))
    db.commit(); db.close()
    return jsonify(ev)
