0. Project Overview
The Tamil Dialogue Generator & Sentence Builder is an AI-powered educational web application. Learners provide grammar inputs — noun (பெயர்ச்சொல்), verb (வினைச்சொல்), tense (காலம்), gender (பால்), and sentence type — and the system generates grammatically correct Tamil sentences and real-life dialogues instantly.

The project is built in 5 progressive phases. Each phase adds new features on top of the last. You do not need prior experience with NLP or databases — every step is explained from scratch.

1.4 Tamil Grammar — Key Concept
Tamil uses Subject-Object-Verb (SOV) word order, opposite to English:

Language
Order
Example
English (SVO)
I + drink + milk
I drink milk
Tamil (SOV)
நான் + பால் + குடிக்கிறேன்
Subject Object Verb

Verb endings change based on tense AND gender/person. This table shows present tense endings:

Person / Gender
Subject
Present Suffix
Example
First Person (I)
நான்
கிறேன்
நான் குடிக்கிறேன் — I drink
Male (He)
அவன்
கிறான்
அவன் குடிக்கிறான் — He drinks
Female (She)
அவள்
கிறாள்
அவள் குடிக்கிறாள் — She drinks
Neutral (It)
அது
கிறது
அது குடிக்கிறது — It drinks
Plural (They)
அவர்கள்
கிறார்கள்
அவர்கள் குடிக்கிறார்கள் — They drink

1.5 grammar_engine.py — Full Code
Create backend/grammar_engine.py:

# backend/grammar_engine.py
class TamilGrammarEngine:
    def __init__(self):
        self.present = {'first':'கிறேன்','male':'கிறான்','female':'கிறாள்','neutral':'கிறது','plural':'கிறார்கள்'}
        self.past    = {'first':'னேன்','male':'னான்','female':'னாள்','neutral':'னது','plural':'னார்கள்'}
        self.future  = {'first':'வேன்','male':'வான்','female':'வாள்','neutral':'கும்','plural':'வார்கள்'}
        self.subjects= {'first':'நான்','male':'அவன்','female':'அவள்','neutral':'அது','plural':'அவர்கள்'}

    def get_suffix(self, tense, gender):
        tmap = {'present':self.present,'past':self.past,'future':self.future}
        return tmap.get(tense, self.present).get(gender, 'கிறேன்')

    def generate_statement(self, noun, verb, tense, gender):
        subj   = self.subjects.get(gender,'நான்')
        suffix = self.get_suffix(tense, gender)
        return f'{subj} {noun} {verb}{suffix}'

    def generate_question(self, noun, verb, tense, gender):
        stmt = self.generate_statement(noun, verb, tense, gender)
        return stmt[:-1] + 'ஆ?'

    def generate_dialogue(self, noun, verb, tense, gender):
        return [
            {'speaker':'நண்பன்', 'line': self.generate_question(noun,verb,tense,gender)},
            {'speaker':'நான்',   'line': self.generate_statement(noun,verb,tense,gender)},
        ]

# backend/app.py
from flask import Flask, request, jsonify, render_template
from flask_cors import CORS
import sys, os
sys.path.insert(0, os.path.dirname(__file__))
from grammar_engine import TamilGrammarEngine

app    = Flask(__name__, template_folder='../frontend/templates',
               static_folder='../frontend/static')
CORS(app)
engine = TamilGrammarEngine()

@app.route('/')
def index(): return render_template('index.html')

@app.route('/generate', methods=['POST'])
def generate():
    d     = request.get_json()
    noun  = d.get('noun','');  verb  = d.get('verb','')
    tense = d.get('tense','present'); gender = d.get('gender','first')
    stype = d.get('sentence_type','statement')
    if stype == 'statement':  out = engine.generate_statement(noun,verb,tense,gender)
    elif stype == 'question': out = engine.generate_question(noun,verb,tense,gender)
    else:                     out = engine.generate_dialogue(noun,verb,tense,gender)
    return jsonify({'type':stype,'output':out})

if __name__ == '__main__': app.run(debug=True, port=5000)

async function generateSentence() {
  const noun  = document.getElementById('noun').value.trim();
  const verb  = document.getElementById('verb').value.trim();
  const tense = document.getElementById('tense').value;
  const gender       = document.getElementById('gender').value;
  const sentence_type = document.getElementById('sentence_type').value;
  if (!noun || !verb) {
    document.getElementById('result').textContent = 'Please enter noun and verb.';
    return; }
  const res  = await fetch('/generate', { method:'POST',
    headers:{'Content-Type':'application/json'},
    body: JSON.stringify({noun,verb,tense,gender,sentence_type}) });
  const data = await res.json();
  if (data.type === 'dialogue') {
    let html = '<strong>உரையாடல்:</strong><br>';
    data.output.forEach(d => { html += `<b>${d.speaker}:</b> ${d.line}<br>`; });
    document.getElementById('result').innerHTML = html;
  } else { document.getElementById('result').textContent = data.output; }
}

# backend/init_db.py  — run: python init_db.py
import sqlite3, os
DB = os.path.join(os.path.dirname(__file__), 'tamil_nlp.db')

def run():
    if os.path.exists(DB):
        print('DB already exists. Delete tamil_nlp.db to re-init.'); return
    conn = sqlite3.connect(DB); c = conn.cursor()
    c.execute('''CREATE TABLE vocabulary(
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        tamil_word TEXT, english_meaning TEXT, category TEXT, gender_class TEXT)''')
    c.execute('''CREATE TABLE grammar_rules(
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        tense TEXT, gender TEXT, suffix TEXT, subject TEXT)''')
    c.execute('''CREATE TABLE saved_sentences(
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        sentence TEXT, sentence_type TEXT, tense TEXT,
        saved_at DATETIME DEFAULT CURRENT_TIMESTAMP)''')
    words = [
        ('பால்','milk','noun','neutral'),('சோறு','rice','noun','neutral'),
        ('தண்ணீர்','water','noun','neutral'),('புத்தகம்','book','noun','neutral'),
        ('பள்ளி','school','noun','neutral'),('வீடு','house','noun','neutral'),
        ('குடி','drink','verb','neutral'),('சாப்பிடு','eat','verb','neutral'),
        ('படி','study','verb','neutral'),('ஓடு','run','verb','neutral'),
        ('பேசு','speak','verb','neutral'),('பார்','see','verb','neutral'),
    ]
    c.executemany('INSERT INTO vocabulary VALUES(NULL,?,?,?,?)', words)
    rules = [
        ('present','first','கிறேன்','நான்'),('present','male','கிறான்','அவன்'),
        ('present','female','கிறாள்','அவள்'),('present','neutral','கிறது','அது'),
        ('present','plural','கிறார்கள்','அவர்கள்'),
        ('past','first','னேன்','நான்'),('past','male','னான்','அவன்'),
        ('past','female','னாள்','அவள்'),('past','neutral','னது','அது'),
        ('past','plural','னார்கள்','அவர்கள்'),
        ('future','first','வேன்','நான்'),('future','male','வான்','அவன்'),
        ('future','female','வாள்','அவள்'),('future','neutral','கும்','அது'),
        ('future','plural','வார்கள்','அவர்கள்'),
    ]
    c.executemany('INSERT INTO grammar_rules VALUES(NULL,?,?,?,?)', rules)
    conn.commit(); conn.close()
    print('Database created: tamil_nlp.db')

if __name__ == '__main__': run()

2.4 db_manager.py — All DB Functions
Create backend/db_manager.py:

# backend/db_manager.py
import sqlite3, os
DB = os.path.join(os.path.dirname(__file__), 'tamil_nlp.db')

def conn():
    c = sqlite3.connect(DB); c.row_factory = sqlite3.Row; return c

def get_all_vocabulary():
    db = conn()
    rows = db.execute('SELECT * FROM vocabulary ORDER BY category').fetchall()
    db.close(); return [dict(r) for r in rows]

def search_vocabulary(q):
    db = conn()
    rows = db.execute('SELECT * FROM vocabulary WHERE tamil_word LIKE ? OR english_meaning LIKE ?',
                      (f'%{q}%',f'%{q}%')).fetchall()
    db.close(); return [dict(r) for r in rows]

def get_suffix(tense, gender):
    db = conn()
    r = db.execute('SELECT suffix FROM grammar_rules WHERE tense=? AND gender=?',
                   (tense,gender)).fetchone()
    db.close(); return r['suffix'] if r else 'கிறேன்'

def get_subject(gender):
    db = conn()
    r = db.execute('SELECT subject FROM grammar_rules WHERE gender=? LIMIT 1',
                   (gender,)).fetchone()
    db.close(); return r['subject'] if r else 'நான்'

def save_sentence(sentence, stype, tense):
    db = conn()
    db.execute('INSERT INTO saved_sentences(sentence,sentence_type,tense) VALUES(?,?,?)',
               (sentence,stype,tense))
    db.commit(); db.close()

def get_saved():
    db = conn()
    rows = db.execute('SELECT * FROM saved_sentences ORDER BY saved_at DESC').fetchall()
    db.close(); return [dict(r) for r in rows]

def delete_saved(sid):
    db = conn(); db.execute('DELETE FROM saved_sentences WHERE id=?',(sid,))
    db.commit(); db.close()

PHASE 3  Voice FeaturesText-to-Speech with gTTS · Browser Speech Recognition · Audio playback

🚀 PHASE: Goal: Users can HEAR the Tamil sentence spoken aloud and optionally speak their input using the microphone.

3.1 How Tamil TTS Works
gTTS (Google Text-to-Speech) takes Tamil text and returns an MP3 audio file. The browser then plays it. The full flow:

Browser  →  POST /speak { text: 'நான் பால் குடிக்கிறேன்' }
Flask    →  gTTS generates MP3  →  saves to static/audio/output.mp3
Browser  ←  receives file path  →  plays with <audio> tag

3.2 Add Audio Folder
mkdir frontend/static/audio

3.3 Add /speak Route to app.py
Add these imports and route to backend/app.py (add to existing file, before the last line):

from gtts import gTTS
import tempfile

AUDIO_DIR = os.path.join(os.path.dirname(__file__), '..', 'frontend', 'static', 'audio')
os.makedirs(AUDIO_DIR, exist_ok=True)

@app.route('/speak', methods=['POST'])
def speak():
    text = request.get_json().get('text','')
    if not text: return jsonify({'error':'no text'}), 400
    out  = os.path.join(AUDIO_DIR, 'output.mp3')
    tts  = gTTS(text=text, lang='ta')   # 'ta' = Tamil
    tts.save(out)
    return jsonify({'audio_url': '/static/audio/output.mp3'})

async function speakSentence() {
  if(!last) return;

    ? last.data.output.map(d=>d.line).join(' ') : last.data.output;
  const res  = await fetch('/speak',{method:'POST',
    headers:{'Content-Type':'application/json'},
    body:JSON.stringify({text})});
  const data = await res.json();
  if(data.audio_url){
    const audio = document.getElementById('audio');
    audio.src = data.audio_url + '?t=' + Date.now(); // avoid cache
    document.getElementById('audioPlayer').style.display='block';
    audio.play();
  }
}

  rec.lang = 'ta-IN';  // Tamil India

@app.route('/chat', methods=['POST'])
def chat():
    d        = request.get_json()
    user_msg = d.get('message','')
    history  = d.get('history', [])
    system   = '''You are a friendly Tamil language tutor named 'Amma'.
You help beginners learn Tamil grammar and sentences.
Reply in simple English with Tamil examples.
Always include the Tamil script alongside transliteration.
Keep answers short — 2 to 4 sentences max.'''
    messages = history + [{'role':'user','content':user_msg}]
    msg = ai_client.messages.create(
        model='claude-sonnet-4-20250514',
        max_tokens=300, system=system, messages=messages
    )
    reply = msg.content[0].text
    return jsonify({'reply': reply})

const CACHE = 'tamil-nlp-v1';
const ASSETS = ['/', '/static/css/style.css', '/static/js/main.js'];
self.addEventListener('install', e => {
  e.waitUntil(caches.open(CACHE).then(c => c.addAll(ASSETS)));
});
self.addEventListener('fetch', e => {
  e.respondWith(caches.match(e.request).then(r => r || fetch(e.request)));
});
